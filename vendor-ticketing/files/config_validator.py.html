# config_validator.py
# Validates vendor config dictionaries before attempting ticket submission.
# Prevents missing required fields from causing silent failures or runtime errors.
# Use this as a reusable logic block across different vendor types.

def validate_vendor_config(config, required_fields):
    """
    Ensures that all required fields exist and are non-empty in the vendor config.
    :param config: dict containing vendor config values
    :param required_fields: list of strings (field names expected)
    :return: True if valid, False otherwise
    """
    missing = []
    for field in required_fields:
        if field not in config or not config[field]:
            missing.append(field)
    if missing:
        print(f"Missing required config fields: {', '.join(missing)}")
        return False
    return True
